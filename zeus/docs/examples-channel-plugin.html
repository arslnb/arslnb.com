server/plugin.py

from __future__ import annotations

import json
import urllib.request
from dataclasses import dataclass


@dataclass
class AcmeAdapter:
    endpoint_base: str
    verify_token: str
    send_api_key: str

    def verify_webhook(self, request):
        mode = str(request.query_params.get("hub.mode") or "")
        token = str(request.query_params.get("hub.verify_token") or "")
        challenge = str(request.query_params.get("hub.challenge") or "")
        if mode == "subscribe" and token == self.verify_token:
            return challenge
        return {"status_code": 403, "body": "forbidden"}

    def ingest_webhook(self, *, request, body: bytes):
        try:
            payload = json.loads(body.decode("utf-8"))
        except Exception:
            return []

        event_id = str(payload.get("id") or "")
        text = str(payload.get("text") or "")
        from_user = str(payload.get("from") or "")
        if not event_id or not from_user:
            return []

        return [
            {
                "message_id": f"acme:{event_id}",
                "channel": "acme",
                "from_user": from_user,
                "thread_hint": from_user,
                "text": text,
                "metadata": {
                    "reply_context": {
                        "to": from_user,
                    },
                    "processing_context": {
                        "to": from_user,
                        "typing": True,
                    }
                },
                "payload": payload,
                "received_at": payload.get("timestamp")
            }
        ]

    def send(self, payload: dict):
        to_value = str(payload.get("to") or "").strip()
        text = str(payload.get("text") or "")
        if not to_value or not text:
            return {"ok": False, "error": "missing_to_or_text"}

        req = urllib.request.Request(
            f"{self.endpoint_base.rstrip('/')}/messages/send",
            method="POST",
            headers={
                "content-type": "application/json",
                "authorization": f"Bearer {self.send_api_key}",
            },
            data=json.dumps({"to": to_value, "text": text}).encode("utf-8"),
        )
        with urllib.request.urlopen(req, timeout=20) as resp:
            body = json.loads(resp.read().decode("utf-8"))

        return {
            "ok": True,
            "provider_message_id": str(body.get("id") or "")
        }


class AcmeChannelPlugin:
    def __init__(self, ctx):
        config = dict(ctx.config or {})
        secrets = dict(getattr(ctx, "secrets", {}) or {})
        verify_token = str(secrets.get("verify_token") or "")
        send_api_key = str(secrets.get("send_api_key") or "")
        owner_handle = str(config.get("owner_handle") or "").strip().lower().replace("@", "")
        self.custom_route = str(config.get("custom_route") or "").strip().lower()
        self.adapter = AcmeAdapter(
            endpoint_base=str(config.get("endpoint_base") or ""),
            verify_token=verify_token,
            send_api_key=send_api_key,
        )

    def channels(self):
        return {"acme": self.adapter}

    def webhooks(self):
        route = self.custom_route or "acme-events"
        return {
            route: {
                "methods": ["POST"],
                "handler": self.handle_custom_event,
            }
        }

    def actions(self):
        return {
            "whoami": self.whoami,
        }

    async def whoami(self, payload=None, context=None):
        del payload, context
        return {"ok": True, "text": "acme action result"}

    def handle_custom_event(self, request):
        body = request.body if hasattr(request, "body") else b""
        payload = {}
        try:
            payload = json.loads(body.decode("utf-8") or "{}")
        except Exception:
            payload = {"raw": body.decode("utf-8", errors="replace")}

        return {
            "message_id": "acme:webhook",
            "channel": "acme",
            "from_user": "system",
            "text": "acme custom event",
            "payload": payload,
        }


def register(ctx):
    return AcmeChannelPlugin(ctx)

In a real channel adapter, use the saved owner_handle (or similar fields) to decide when to stamp metadata.sender_class = "owner" on inbound envelopes. Zeus renders those fields directly under Settings > Messaging Channels when you declare the ui_hints.messaging_channels.<channel>.primary_user_identity block.

If the user turns on Offline replies while Mac is offline, Zeus may temporarily answer from the server when the Mac does not claim the turn quickly enough. Channel plugins do not need a second webhook contract for this; they only need stable conversation_key identity and usable reply_context so the fallback and the later Mac resume both stay on the same remote conversation.

Install and Activate

POST /v1/client/plugins/install
{
  "plugin_id": "acme_channel",
  "version": "1.0.0",
  "config": {
    "server": {
      "endpoint_base": "https://api.acme.dev",
      "custom_route": "stripe-events"
    }
  },
  "secret_refs": {
    "server": {
      "verify_token": "acme_verify_token",
      "send_api_key": "acme_send_api_key"
    }
  }
}

After success, these routes are live:

  • GET /v1/channels/acme/webhook
  • POST /v1/channels/acme/webhook
  • POST /v1/client/channels/acme/send
  • POST /v1/client/plugins/acme_channel/actions/whoami
  • POST /webhook/stripe-events

Real Registry Reference

If you want a working sample instead of the synthetic acme_channel example, look at the registry plugin kitchen_sink_plugin. It implements a real Telegram channel path at /v1/channels/telegram/webhook, auto-registers its Telegram webhook when ZEUS_PUBLIC_BASE_URL is configured, exposes owner-identity fields in the macOS Settings > Messaging Channels UI, uses signed server actions to power the daemon-side github_me and github_notifications tools over the stored server_github OAuth session, and stays compatible with the opt-in offline fallback reply flow because it emits stable canonical conversation identity plus reply context.

Add a client component if you also need daemon tools/hooks.
